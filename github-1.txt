import streamlit as st
from transformers import pipeline
@st.cache_resource
def load_model():
    return pipeline("sentiment-analysis")
nlp = load_model()
st.set_page_config(page_title="SentimentSense", page_icon="😊", layout="centered")
st.title("📊 SentimentSense: NLP Powered Sentiment Analysis")
st.write("Type your sentence below and find out if it's Positive, Negative, or Neutral!")
user_input = st.text_area("✍️ Enter your text here:")

if st.button("Analyze Sentiment"):
    if user_input.strip():
        with st.spinner("Analyzing..."):
            result = nlp(user_input)[0]
            st.success(f"**Sentiment:** {result['label']}  \n**Confidence:** {result['score']:.2f}")
    else:
        st.warning("⚠️ Please enter some text first.")
